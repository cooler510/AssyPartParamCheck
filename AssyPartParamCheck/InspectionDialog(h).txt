#pragma once

#include <QDialog>
#include <QLabel>
#include <QPushButton>
#include <QString>
#include <QTabWidget>
#include <QTableWidget>
#include <QTextEdit>
#include <QVBoxLayout>
#include <vector>

#include <ProObjects.h>

// 标准件信息
struct StandardPartInfo
{
    ProMdl mdl;       // Creo 模型句柄，用于回写参数
    QString fileName; // 图号：文件名.prt
    QString sname;    // 标准件名称
    QString stype;    // 型号
    QString vendor;   // 供应商
};

// 普通零件信息
struct PartInfo
{
    ProMdl mdl;       // Creo 模型句柄，用于回写参数
    QString fileName; // 图号：文件名.prt
    QString dname;    // 零件名称
    QString finish;   // 表面处理
    QString material; // 材料
    QString qty;      // 零件数量
};

// 未分类零件信息：没有任何标识参数，无法判断是标准件还是普通零件
struct UnknownPartInfo
{
    ProMdl mdl;         // Creo 模型句柄
    QString fileName;   // 图号：文件名.prt
    QString suggestion; // 建议归类说明
    int chosenType = 0; // 用户在未分类页选的类型：0=标准件, 1=零件（每行独立，不参与全局视图）
};

class InspectionDialog : public QDialog
{
    Q_OBJECT

public:
    explicit InspectionDialog(QWidget *parent = nullptr);

private slots:
    void refreshTables();
    void saveChanges();
    void onItemChanged(QTableWidgetItem *item);

private:
    void collectData();
    void setupTable(QTableWidget *table);
    void applyUnknownColumnView();

    QTabWidget *m_tabWidget;
    QTableWidget *m_standardTable;
    QTableWidget *m_partTable;
    QTableWidget *m_unknownTable;
    QTextEdit *m_debugEdit;
    QLabel *m_statusLabel;
    QPushButton *m_saveBtn;

    std::vector<StandardPartInfo> m_standardParts;
    std::vector<PartInfo> m_parts;
    std::vector<UnknownPartInfo> m_unknownParts;
    QString m_debugText;
};
