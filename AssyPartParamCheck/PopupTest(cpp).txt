// 使用 show() 而非 exec()，避免嵌套事件循环导致 Creo 损坏
static int PartInspectionFunc(uiCmdCmdId /*command*/, uiCmdValue * /*p_value*/, void * /*p_push_command_data*/)
{
    // 防止重复打开多个对话框
    static InspectionDialog *s_dialog = nullptr;
    if (s_dialog)
    {
        s_dialog->raise();
        s_dialog->activateWindow();
        s_dialog->showNormal();
        return 0;
    }

    s_dialog = new InspectionDialog();
    s_dialog->setAttribute(Qt::WA_DeleteOnClose);

    // 对话框销毁时清空静态指针，避免悬空指针
    QObject::connect(s_dialog, &QObject::destroyed, [](QObject *)
                     { s_dialog = nullptr; });

    // show() 而非 exec()：不阻塞 Creo 主线程
    s_dialog->show();

    return 0;
}

// 注册零件参数检查菜单
void RegisterPartInspection()
{
    ProError status;

    // 在 Help 菜单后添加"检查"菜单（菜单显示名在 IconMessage.txt 中映射）
    // 注意: ProMenubarMenuAdd 的 filename 参数是 ProFileName(wchar_t*)，不是 char*
    status = ProMenubarMenuAdd((char *)"PopupTest", (char *)"PopupTest",
                               (char *)"Help", PRO_B_FALSE, (wchar_t *)L"IconMessage.txt");
    if (status != PRO_TK_NO_ERROR)
        return;

    // 检查零件参数按钮
    // ProCmdActionAdd 签名: (action_name, action_cb, priority, access_func,
    //                        allow_in_non_active_window, allow_in_accessory_window, action_id)
    // 没有 filename 参数
    uiCmdCmdId cmdId;
    status = ProCmdActionAdd((char *)"PartInspectionAct",
                             (uiCmdCmdActFn)PartInspectionFunc,
                             uiProe2ndImmediate, nullptr, PRO_B_FALSE, PRO_B_TRUE,
                             &cmdId);
    if (status != PRO_TK_NO_ERROR)
        return;

    status = ProMenubarmenuPushbuttonAdd((char *)"PopupTest",
                                         (char *)"PartInspectionLabel",
                                         (char *)"PartInspectionLabel",
                                         (char *)"PartInspectionTips",
                                         nullptr, PRO_B_TRUE, cmdId,
                                         (wchar_t *)L"IconMessage.txt");
}
